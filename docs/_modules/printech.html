def table(column_names, rows, title='', show_lines=False):
    table = Table(title=title, show_lines=show_lines)

    # Add columns
    for column in column_names:
        table.add_column(column)

    # Add rows
    for row in rows:
        table.add_row(*row)

    console.print(table, style="green")
